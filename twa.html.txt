            { name: "brand", weight: 0.15 },
            { name: "category", weight: 0.1 },
          ],
          threshold: 0.3,
          distance: 100,
          ignoreLocation: true,
          minMatchCharLength: 2,
        });
        fuseSig = sig;
        return fuseInstance;
      };

                category: info.category ?? item.category,

        });

          let ean13 = 0;

            if (i.code.length === 8) ean8++;
            else ean13++;
          }
          this._statsCache = {
            total: items.length,
            categories: cats.size,
            ean8,
            ean13,

          this._statsSig = sig;
          return this._statsCache;

        async _bootstrap() {
          if (this._booted) return;
          this._booted = true;

          const startedAt = performance.now();
          const cloudItems = await cloudLoad();
          if (Array.isArray(cloudItems)) {
            const decorated = cloudItems.map(sanitizeRecord).filter(Boolean);
            const collapsed = mergeItems(decorated, []);
            store().items = collapsed;
            if (signature(collapsed) !== signature(decorated)) {
              persistQueue.add(() => persistItems(store().items.slice()));
            }
            this._hydrateInfo(collapsed);
          }

          const elapsed = performance.now() - startedAt;
          if (elapsed < SKELETON_DURATION)
            await new Promise((r) =>
              setTimeout(r, SKELETON_DURATION - elapsed),
            );

          store().isLoading = false;

          this.$watch("items", () => {
            this._updateClosingConfirmation();
            this._validateActiveTab();
            this.$nextTick(() => this._initSwipeGestures());
          });
          this._updateClosingConfirmation();

          this._clockInterval = setInterval(() => {
            const c = Alpine.store("clock");
            c.now = Date.now();
            const today = dayjs().format("YYYY-MM-DD");
            if (today !== c.day) {
              c.day = today;
              this._validateActiveTab();
            }
          }, 30_000);

          this._searchDebounced = debounce((val) => {
            this.debouncedQuery = val;
          }, 200);
          this.$watch("searchQuery", (val) => this._searchDebounced(val));

          const easing = getComputedStyle(document.documentElement)
            .getPropertyValue("--motion-standard")
            .trim();
          if (easing) this._easing = easing;

          this.$nextTick(() => this._initSwipeGestures());

        _initSwipeGestures() {
          this._swipeObserver?.disconnect();
          this._swipeObserver = null;

          const list = document.querySelector(".item-list");
          if (!list) {
            this._destroyGestures();
            return;
          }

          const easing = this._easing;
          const removeItem = (id) => this.removeItem(id);
          const gestures = this._gestures;

          const attach = (wrapper) => {
            if (wrapper.hasAttribute("data-gesture")) return;
            wrapper.setAttribute("data-gesture", "");
            const card = wrapper.querySelector(".barcode-card");
            if (!card) return;
            const gesture = new TinyGesture(card, {
              threshold: (type) => (type === "x" ? 25 : Infinity),
            });
            gestures.add(gesture);
            let offset = 0;

            gesture.on("panmove", () => {
              if (gesture.touchMoveX === null) return;
              offset = Math.min(0, gesture.touchMoveX);
              card.classList.add("barcode-card--swiping");
              card.style.transform = `translateX(${offset}px)`;
              card.style.transition = "none";
            });

            gesture.on("panend", () => {
              card.style.transition = `transform 200ms ${easing}`;
              card.classList.remove("barcode-card--swiping");
              if (offset < -80) {
                const id = wrapper.dataset.id;
                if (id) removeItem(id);
              } else {
                card.style.transform = "";
              }
              offset = 0;
            });
          };

          this._swipeObserver = new MutationObserver(() => {
            for (const g of [...gestures]) {
              if (g.element && !g.element.isConnected) {
                try {
                  g.destroy();
                } catch {}
                gestures.delete(g);
              }
            }
            list
              .querySelectorAll(".swipe-wrapper:not([data-gesture])")
              .forEach(attach);
          });
          this._swipeObserver.observe(list, { childList: true, subtree: true });
          list.querySelectorAll(".swipe-wrapper").forEach(attach);
        },

        _processScan(raw) {
          const parsed = parseCEN(raw);

          if (!parsed) {
            notify("error");
            haptic("medium");
            if (this.isMultiScan) return false;
            showToast("Поддерживаются только коды в формате CEN", "err");
            this._closeScanner();
            return true;
          }

          const incomingPrice = formatPrice(parsed.price);
          const incomingTicketTs = ticketDateToTs(parsed.date);
          const existing = store().items.find((i) => i.code === parsed.code);

          if (!existing) {
            notify("success");
            haptic();
            playSuccessSound();
            this._insertItem(parsed, incomingPrice, incomingTicketTs);
            if (this.isMultiScan) return false;
            showToast(`Код ${parsed.code} добавлен`);
            this._closeScanner();
            return true;
          }

          const sameDate =
            (existing.ticketTs ?? null) === (incomingTicketTs ?? null);
          const samePrice =
            (existing.price ?? null) === (incomingPrice ?? null);
          const changed = this._updateItem(
            existing,
            parsed,
            incomingPrice,
            incomingTicketTs,
          );

          if (!changed && sameDate && samePrice) {
            notify("warning");
            haptic("medium");
            if (this.isMultiScan) return false;
            showToast(`Код ${parsed.code} уже в списке`, "warn");
            this._closeScanner();
            return true;
          }

          notify("success");
          haptic();
          playSuccessSound();
          if (this.isMultiScan) return false;
          showToast(`Карточка ${parsed.code} обновлена`);
          this._closeScanner();
          return true;
        },

        _updateItem(existing, parsed, incomingPrice, incomingTicketTs) {
          const now = Date.now();
          const history = (existing.history ?? []).slice();
          let changed = false;

          if (incomingPrice && parsed.date && incomingTicketTs != null) {
            const idx = history.findIndex(
              (h) => h.ticketTs === incomingTicketTs,
            );
            if (idx === -1) {
              history.push({
                price: incomingPrice,
                date: parsed.date,
                ticketTs: incomingTicketTs,
                ts: now,
              });
              changed = true;
            } else if (history[idx].price !== incomingPrice) {
              history[idx] = {
                price: incomingPrice,
                date: parsed.date,
                ticketTs: incomingTicketTs,
                ts: now,
              };
              changed = true;
            }
          }

          history.sort((a, b) => a.ticketTs - b.ticketTs);
          const latest = history.at(-1) ?? null;

          const priceChanged =
            (existing.price ?? null) !==
            (latest ? latest.price : existing.price);
          const dateChanged =
            (existing.date ?? null) !== (latest ? latest.date : existing.date);
          const rawChanged = (existing.raw ?? null) !== parsed.raw;
          if (priceChanged || dateChanged || rawChanged) changed = true;

          if (!changed) return false;

          store().items = store()
            .items.map((i) =>
              i.id === existing.id
                ? {
                    ...i,
                    raw: parsed.raw,
                    history,
                    price: latest ? latest.price : i.price,
                    date: latest ? latest.date : i.date,
                    ticketTs: latest ? latest.ticketTs : i.ticketTs,
                    ts: now,
                  }
                : i,
            )
            .sort((a, b) => b.ts - a.ts || (a.id < b.id ? 1 : -1));
          persistQueue.add(() => persistItems(store().items.slice()));
          this._flashCard(existing.id);
          return true;
        },

        _flashCard(id) {
          this.$nextTick(() => {
            const card = document.querySelector(
              `.barcode-card[data-card-id="${cssEscape(id)}"]`,
            );
            card?.animate(
              [
                {
                  boxShadow:
                    "0 0 0 0 color-mix(in srgb, var(--color-primary) 55%, transparent)",
                },
                { boxShadow: "0 0 0 8px transparent" },
              ],
              { duration: 650, easing: this._easing },
            );
          });
        },

        _insertItem(parsed, incomingPrice, incomingTicketTs) {
          const now = Date.now();
          const history =
            incomingPrice && parsed.date && incomingTicketTs != null
              ? [
                  {
                    price: incomingPrice,
                    date: parsed.date,
                    ticketTs: incomingTicketTs,
                    ts: now,
                  },
                ]
              : [];
          const item = createItem({
            code: parsed.code,
            raw: parsed.raw,
            price: incomingPrice,
            date: parsed.date,
            ticketTs: incomingTicketTs,
            history,
            ts: now,
          });
          store().items = [item, ...store().items];
          persistQueue.add(() => persistItems(store().items.slice()));
          resolveInfo(item.id, parsed.code);
        },

        async removeItem(id) {
          haptic("medium");
          const wrapper = document.querySelector(
            `.swipe-wrapper[data-id="${cssEscape(id)}"]`,
          );
          const card = wrapper?.querySelector(".barcode-card");
          if (card) {
            try {
              await card.animate(
                [
                  { opacity: 1, transform: "translateX(0) scale(1)" },
                  { opacity: 0, transform: "translateX(-30px) scale(0.95)" },
                ],
                { duration: 180, easing: this._easing, fill: "forwards" },
              ).finished;
            } catch {}
          }
          store().items = store().items.filter((i) => i.id !== id);
          persistQueue.add(() => persistItems(store().items.slice()));
        },

        async clearAll() {
          haptic("medium");
          if (
            !(await confirmDialog("Удалить все записи?\nДействие необратимо."))
          )
            return;

          const wrappers = [
            ...document.querySelectorAll(".swipe-wrapper[data-id]"),
          ];
          wrappers.forEach((el, i) => {
            el.style.animation = `card-out 120ms var(--motion-standard) ${i * 20}ms forwards`;
          });
          if (wrappers.length)
            await new Promise((r) => setTimeout(r, wrappers.length * 20 + 120));

          this.activeTab = "all";
          store().items = [];
          persistQueue.add(() => persistItems([]));
          showToast("История очищена");
          notify("success");
        },

        async copyText(text) {
          haptic();
          let success = false;
          if (navigator.clipboard?.writeText) {
            try {
              await navigator.clipboard.writeText(text);
              success = true;
            } catch {}
          }
          showToast(
            success ? "Скопировано в буфер обмена" : "Не удалось скопировать",
            success ? "ok" : "err",
          );
          notify(success ? "success" : "error");
        },

        downloadBarcode(item) {
          haptic();
          if (!supports("8.0") || !tg.downloadFile) {
            showToast(
              "Скачивание не поддерживается этой версией Telegram",
              "err",
            );
            return;
          }
          tg.downloadFile({
            url: `https://barcodeapi.org/api/${item.code.length === 8 ? "EAN8" : "EAN13"}/${item.code}`,
            file_name: `${item.code}.png`,
          });
        },
      }));

      Alpine.start();
